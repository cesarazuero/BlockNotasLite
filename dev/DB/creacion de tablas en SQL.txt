  `id` INTEGER NULL AUTO_INCREMENT DEFAULT NULL,
  `id ARCHIVO` INTEGER NULL DEFAULT NULL,
  `Nombre` VARCHAR(50) NULL DEFAULT NULL,
  PRIMARY KEY (`id`)

-- ---
-- Table 'TEXTOS'
-- 
-- ---

DROP TABLE IF EXISTS `TEXTOS`;
		
CREATE TABLE `TEXTOS` (
  `id` INTEGER NULL AUTO_INCREMENT DEFAULT NULL,
  `texto` VARCHAR(200) NULL DEFAULT NULL,

  `posY` DOUBLE NULL DEFAULT NULL,
  `ancho` FLOAT NULL DEFAULT NULL,
  `alto` FLOAT NULL DEFAULT NULL,
  `escalaX` FLOAT NULL DEFAULT NULL,
  `escalaY` INTEGER NULL DEFAULT NULL,
  `rotacion` FLOAT NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
);
